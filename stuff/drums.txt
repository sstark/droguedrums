# drougedrums
#
# to play this:
#
# drougedrums drums.txt seqfull
#
# the file can be changed while it is being played.
#
# keywords: set, track, seq
#

# define a drum set:
set: basicset
# lane midichannel notenumber
bd: 5 1  # bass drum
hc: 5 5  # hi hat closed
ho: 5 6  # hi hat opened
sd: 5 4  # snare drum

track: beat44 step=4 bpm=100
hc ho hc hc hc ho hc hc
bd          bd
      sd          sd

track: beat68 step=8 bpm=50
hc hc hc \
hc hc hc \
hc hc hc \
hc hc hc

track: break

seq: straightseq
4 x beat44

seq: breakseq
1 x break

seq: oddseq

seq: full
2 x straightseq
1 x breakseq
2 x oddseq
1 x breakseq
repeat
